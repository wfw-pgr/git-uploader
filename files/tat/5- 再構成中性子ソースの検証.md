## plot_neutrons.py (修正版)

```python
import os, sys, json5
import numpy  as np
import pandas as pd
import nk_toolkit.plot.load__config   as lcf
import nk_toolkit.plot.gplot1D        as gp1
import matplotlib.pyplot              as plt


# inpFile  = "dat/angle_energy_vs_neutrons.dat"
# inpFile  = "dat/reconstructed_neutrons.dat"
inpFile  = "dat/ebeamDriven_neutrons.dat"
pngDir   = "png/ebeamDriven/"

# ------------------------------------------------- #
# --- [1] load data                             --- #
# ------------------------------------------------- #
df       = pd.read_csv( inpFile, sep=r"\s+" )
sr       = 2.0*np.pi * ( np.cos( df["al"]*np.pi/180.0 ) - np.cos( df["ah"]*np.pi/180.0 ) )
dE       = df["eh"] - df["el"]
df["nc"] = df["n[count/sr/MeV]"] * sr * dE
df_step  = pd.DataFrame({
    "al": df[["al", "al"]].values.ravel(),
    "ah": df[["ah", "ah"]].values.ravel(),
    "E":  df[["el", "eh"]].values.ravel(),
    "n":  df[["nc", "nc"]].values.ravel(),
})

# ------------------------------------------------- #
# --- [2] config (energy)                       --- #
# ------------------------------------------------- #
config  = {
    **lcf.load__config(), 
    "figure.size"        : [4.5,4.5],
    "figure.pngFile"     : pngDir + "energy_vs_neutron__byAngle.png", 
    "figure.position"    : [ 0.16, 0.16, 0.94, 0.94 ],
    "ax1.x.normalize"    : 1.0e-6, 
    "ax1.y.normalize"    : 1.0e0, 
    "ax1.x.range"        : { "auto":False, "min": 1.0e-4, "max":1.0e+8, "num":13 },
    "ax1.y.range"        : { "auto":False, "min": 1.0e+0, "max":1.0e+12, "num":7 },
    "ax1.x.label"        : r"$Energy \ \mathrm{[eV]}$",
    "ax1.y.label"        : r"$neutrons \ \mathrm{[counts]}$",
    "ax1.x.minor.nticks" : 1,
    "ax1.x.log"          : True,
    "ax1.y.log"          : True,
    "plot.linestyle"     : "-",
    "plot.linewidth"     : 1.0,
    "plot.marker"        : "none",
    "plot.markersize"    : 0.0,
    "legend.fontsize"    : 7.0, 
}


# ------------------------------------------------- #
# --- [3] plot data by energy                   --- #
# ------------------------------------------------- #
als = np.linspace( 0, 80, 9, dtype=int )
ahs = [ al+10 for al in als ]
fig = gp1.gplot1D( config=config )
for al,ah in zip(als,ahs):
    df_    = df_step[ ( df_step["al"] == al ) ]
    label  = "{0}-{1} deg.".format( al, ah )
    fig.add__plot( xAxis=df_["E"], yAxis=df_["n"], label=label )
fig.set__axis()
fig.set__legend()
fig.save__figure()


# ------------------------------------------------- #
# --- [4] config (angle)                        --- #
# ------------------------------------------------- #
df_step  = pd.DataFrame({
    "a" : df[["al", "ah"]].values.ravel(),
    "el": df[["el", "el"]].values.ravel(),
    "eh": df[["eh", "eh"]].values.ravel(),
    "n" : df[["nc", "nc"]].values.ravel(),
})

config = {
    **config,
    "figure.pngFile"     : pngDir + "angle_vs_neutron__byEnergy.png", 
    "ax1.x.normalize"    : 1.0e0, 
    "ax1.y.normalize"    : 1.0e0,
    "ax1.x.range"        : { "auto":False, "min": 0.0, "max":100.0, "num":11 },
    "ax1.y.range"        : { "auto":False, "min": 1.0e+0, "max":1.0e+12, "num":7 },
    "ax1.x.label"        : r"$Angle \ \mathrm{[deg.]}$",
    "ax1.y.label"        : r"$neutrons \ \mathrm{[counts]}$",
    "ax1.x.log"          : False,
    "plot.marker"        : "o",
    "plot.markersize"    : 1.0,
}

# ------------------------------------------------- #
# --- [5] plot data by angle                    --- #
# ------------------------------------------------- #
els    = sorted( list( set( df["el"].to_list() ) ) )
ehs    = sorted( list( set( df["eh"].to_list() ) ) )
fig    = gp1.gplot1D( config=config )
colors = plt.cm.jet( np.linspace( 0.0, 1.0, len(els)+2 ) )[1:-1]
for ik,(el,eh) in enumerate( zip(els,ehs) ):
    df_    = df_step[ ( df_step["el"] == el ) ]
    label  = "E={0}-{1}".format( el, eh )
    fig.add__plot( xAxis=df_["a"], yAxis=df_["n"], label=label, color=colors[ik] )
fig.set__axis()
fig.set__legend()
fig.save__figure()


# ------------------------------------------------- #
# --- [6] config (energy)                       --- #
# ------------------------------------------------- #
config  = {
    **lcf.load__config(), 
    "figure.size"        : [4.5,4.5],
    "figure.pngFile"     : pngDir + "energy_vs_norm_n__byAngle.png", 
    "figure.position"    : [ 0.16, 0.16, 0.94, 0.94 ],
    "ax1.x.normalize"    : 1.0e-6, 
    "ax1.y.normalize"    : 1.0e0, 
    "ax1.x.range"        : { "auto":False, "min": 1.0e-4, "max":1.0e+8, "num":13 },
    "ax1.y.range"        : { "auto":False, "min": 1.0e+0, "max":1.0e+16, "num":9 },
    "ax1.x.label"        : r"$Energy \ \mathrm{[eV]}$",
    "ax1.y.label"        : r"$neutrons \ \mathrm{[counts/sr/MeV]}$",
    "ax1.x.minor.nticks" : 1,
    "ax1.x.log"          : True,
    "ax1.y.log"          : True,
    "plot.linestyle"     : "-",
    "plot.linewidth"     : 1.0,
    "plot.marker"        : "none",
    "plot.markersize"    : 0.0,
    "legend.fontsize"    : 7.0, 
}


# ------------------------------------------------- #
# --- [7] plot data by energy                   --- #
# ------------------------------------------------- #
sr       = 2.0*np.pi * ( np.cos( df["al"]*np.pi/180.0 ) - np.cos( df["ah"]*np.pi/180.0 ) )
dE       = df["eh"] - df["el"]
df["nc"] = df["n[count/sr/MeV]"]
df_step  = pd.DataFrame({
    "al": df[["al", "al"]].values.ravel(),
    "ah": df[["ah", "ah"]].values.ravel(),
    "E":  df[["el", "eh"]].values.ravel(),
    "n":  df[["nc", "nc"]].values.ravel(),
})

als = np.linspace( 0, 80, 9, dtype=int )
ahs = [ al+10 for al in als ]
fig = gp1.gplot1D( config=config )
for al,ah in zip(als,ahs):
    df_    = df_step[ ( df_step["al"] == al ) ]
    label  = "{0}-{1} deg.".format( al, ah )
    fig.add__plot( xAxis=df_["E"], yAxis=df_["n"], label=label )
fig.set__axis()
fig.set__legend()
fig.save__figure()

```
