# DTL要素で使用している電場マップ (eh_DTL.#nn)

- eh_DTL.#nnをもとに作成。
- 足利さんの引き継ぎより、femtetの計算ファイルあり。
	- これによると、円筒状の平板ギャップで電場を作る感じになっている。
	- 模式的に表すと下記 (A-B間で電圧印加)

```
      A |     | B
-----         ----------
   ビームパス
-----         ----------
      A |     | B
```

```python
def colormap__efield():


    # ------------------------------------------------- #
    # --- [1] read data                             --- #
    # ------------------------------------------------- #
    def read__eh_DTL( fname ):
        lines        = [s.strip() for s in open(fname, encoding="utf-8", errors="ignore") if s.strip()]
        ksym         = int( lines[3] )
        rmin,rmax,nr = map(float, lines[4].split()); nr = int(nr)
        zmin,zmax,nz = map(float, lines[5].split()); nz = int(nz)
        df_          = pd.read_csv( fname, sep=r"\s+", skiprows=6, nrows=nr*nz, \
                                    header=None, names=["Er", "Ez", "Et"] )
        if ( ksym == 2 ):
            Er_half = df_["Er"].to_numpy().reshape( (nr, nz) )
            Ez_half = df_["Ez"].to_numpy().reshape( (nr, nz) )
            Er_full = np.concatenate( [ -Er_half[:, ::-1][:, :-1], Er_half], axis=1 )
            Ez_full = np.concatenate( [  Ez_half[:, ::-1][:, :-1], Ez_half], axis=1 )
        else:
            Er_full = Er_half
            Ez_full = Ez_half
        ra, za       = np.linspace( rmin, rmax, nr ), np.linspace( -zmax, zmax, 2*nz-1 )
        rg, zg       = np.meshgrid( ra, za, indexing="ij")
        df           = pd.DataFrame.from_dict( { "rAxis": rg.ravel(), "zAxis": zg.ravel(), \
                                                 "Er":Er_full.ravel(), "Ez":Ez_full.ravel() } )
        return { "ksym": ksym, "rAxis": [rmin,rmax,nr], "zAxis": [zmin,zmax,nz], "df": df }

    fname = "eh_DTL.#01"
    Data  = read__eh_DTL( fname )

    # ------------------------------------------------- #

        "ax1.y.range"        : { "auto":True, "min": 0.0, "max":1.0, "num":11 },
        "ax1.x.label"        : "z (cm)",
        "ax1.y.label"        : "r (cm)",
        "ax1.x.minor.nticks" : 1,
        "cmp.level"          : { "auto":True, "min": 0.0, "max":1.0, "num":11 },
        "cmp.colortable"     : "jet",
    }

    # ------------------------------------------------- #
    # --- [3] plot data                             --- #
    # ------------------------------------------------- #
    config["figure.pngFile"] = "png/Er.png"
    fig                      = gp2.gplot2D( xAxis=df["zAxis"].to_numpy(), yAxis=df["rAxis"].to_numpy(), cMap=df["Er"].to_numpy(), config=config )
    config["figure.pngFile"] = "png/Ez.png"
    fig                      = gp2.gplot2D( xAxis=df["zAxis"].to_numpy(), yAxis=df["rAxis"].to_numpy(), cMap=df["Ez"].to_numpy(), config=config )

    # ------------------------------------------------- #
    # --- [4] plot data (1D)                        --- #
    # ------------------------------------------------- #
    dfz1d = df[ df["rAxis"] == 0.0 ]
    dfr1d = df[ df["zAxis"] == 0.0 ]
    config   = {
        **lcf.load__config(), 
        "figure.size"        : [4.5,4.5],
        "figure.pngFile"     : "png/Erz_on_zAxis.png", 
        "figure.position"    : [ 0.16, 0.16, 0.94, 0.94 ],
        "ax1.y.normalize"    : 1.0e0, 
        "ax1.x.range"        : { "auto":True, "min": 0.0, "max":1.0, "num":11 },
        "ax1.y.range"        : { "auto":True, "min": 0.0, "max":1.0, "num":11 },
        "ax1.x.label"        : "z (cm)",
        "ax1.y.label"        : "E (MV/m)",
        "ax1.x.minor.nticks" : 1, 
        "plot.marker"        : "o",
        "plot.markersize"    : 3.0,
        "legend.fontsize"    : 9.0, 
    }
    fig    = gp1.gplot1D( config=config )
    fig.add__plot( xAxis=dfz1d["zAxis"], yAxis=dfz1d["Er"], label="Er" )
    fig.add__plot( xAxis=dfz1d["zAxis"], yAxis=dfz1d["Ez"], label="Ez" )
    fig.set__axis()
    fig.set__legend()
    fig.save__figure()
    
    config   = {
        **lcf.load__config(), 
        "figure.size"        : [4.5,4.5],
        "figure.pngFile"     : "png/Erz_on_rAxis.png", 
        "figure.position"    : [ 0.16, 0.16, 0.94, 0.94 ],
        "ax1.y.normalize"    : 1.0e0, 
        "ax1.x.range"        : { "auto":True, "min": 0.0, "max":1.0, "num":11 },
        "ax1.y.range"        : { "auto":True, "min": 0.0, "max":1.0, "num":11 },
        "ax1.x.label"        : "r (cm)",
        "ax1.y.label"        : "E (MV/m)",
        "ax1.x.minor.nticks" : 1, 
        "plot.marker"        : "o",
        "plot.markersize"    : 3.0,
        "legend.fontsize"    : 9.0, 
    }
    fig    = gp1.gplot1D( config=config )
    fig.add__plot( xAxis=dfr1d["rAxis"], yAxis=dfr1d["Er"], label="Er" )
    fig.add__plot( xAxis=dfr1d["rAxis"], yAxis=dfr1d["Ez"], label="Ez" )
```

``` fortran
        read(1,*) ksym_x, ksym_y, ksym_z
        read(1,*) xmini, xmaxi, nxx
        read(1,*) ymini, ymaxi, nyy
        read(1,*) zmini, zmaxi, nzz
```

## 可視化コード

```python

# ========================================================= #
# ===  colormap__bfield.py                              === #
# ========================================================= #

def colormap__bfield():

    # ------------------------------------------------- #
    # --- [1] read data                             --- #
    # ------------------------------------------------- #
    def read__eh_SOL( fname ):
        lines = [s.strip() for s in open(fname, encoding="utf-8", errors="ignore") if s.strip()]

        # header
        ksymx, ksymy, ksymz = map(int,   lines[1].split())
        xmin, xmax, nx      = map(float, lines[2].split()); nx = int(nx)
        ymin, ymax, ny      = map(float, lines[3].split()); ny = int(ny)
        zmin, zmax, nz      = map(float, lines[4].split()); nz = int(nz)

        df_      = pd.read_csv( fname, sep=r"\s+", skiprows=5, nrows=nx*ny*nz,
                                header=None, names=["Bx", "By", "Bz"] )
        bx       = df_["Bx"].to_numpy()
        by       = df_["By"].to_numpy()
        bz       = df_["Bz"].to_numpy()
        xa       = np.linspace( xmin, xmax, nx )
        ya       = np.linspace( ymin, ymax, ny )
        za       = np.linspace( zmin, zmax, nz )
        xg,yg,zg = np.meshgrid( xa, ya, za, indexing="ij" )
        
        ret = pd.DataFrame.from_dict( { "xAxis":xg.ravel(),"yAxis":yg.ravel(), "zAxis":zg.ravel(),
                                        "Bx": bx.ravel(), "By": by.ravel(), "Bz": bz.ravel() } )
        return( ret )
    fname = "eh_SOL.#64"
    df    = read__eh_SOL( fname )

    # ------------------------------------------------- #
    # --- [2] config                                --- #
    # ------------------------------------------------- #
    config = {
        **lcf.load__config(),
        "figure.size"        : [4.5,4.5],
        "figure.position"    : [0.16, 0.16, 0.84, 0.84],
        "ax1.y.normalize"    : 1.0e0,
        "ax1.x.range"        : { "auto":True, "min":0.0, "max":1.0, "num":11 },
        "ax1.y.range"        : { "auto":True, "min":0.0, "max":1.0, "num":11 },
        "ax1.x.label"        : "z (cm)",
        "ax1.y.label"        : "x (cm)",
        "ax1.x.minor.nticks" : 1,
        "cmp.level"          : { "auto":True, "min":0.0, "max":1.0, "num":11 },
        "cmp.colortable"     : "jet",
    }

    # ------------------------------------------------- #
    # --- [3] plot data (2D, y=0 plane)            --- #
    # ------------------------------------------------- #
    df2d = df[ df["yAxis"] == 0.0 ]

    config["figure.pngFile"] = "png/Bx_xz.png"
    fig = gp2.gplot2D( xAxis=df2d["zAxis"].to_numpy(), yAxis=df2d["xAxis"].to_numpy(),
                       cMap =df2d["Bx"].to_numpy(), config=config )

    config["figure.pngFile"] = "png/By_xz.png"
    fig = gp2.gplot2D( xAxis=df2d["zAxis"].to_numpy(), yAxis=df2d["xAxis"].to_numpy(),
                       cMap =df2d["By"].to_numpy(), config=config )

    config["figure.pngFile"] = "png/Bz_xz.png"
    fig = gp2.gplot2D( xAxis=df2d["zAxis"].to_numpy(), yAxis=df2d["xAxis"].to_numpy(),
                       cMap =df2d["Bz"].to_numpy(), config=config )

    # ------------------------------------------------- #
    # --- [4] plot data (1D, on-axis)               --- #
    # ------------------------------------------------- #
    df1d   = df[ (df["xAxis"] == 0.0) & (df["yAxis"] == 0.0) ]
    config = {
        **lcf.load__config(),
        "figure.size"        : [4.5,4.5],
        "figure.pngFile"     : "png/Bxyz_on_axis.png",
        "figure.position"    : [0.16, 0.16, 0.94, 0.94],
        "ax1.y.normalize"    : 1.0e0,
        "ax1.x.range"        : { "auto":True, "min":0.0, "max":1.0, "num":11 },
        "ax1.y.range"        : { "auto":True, "min":0.0, "max":1.0, "num":11 },
        "ax1.x.label"        : "z (cm)",
        "ax1.y.label"        : "B",
        "ax1.x.minor.nticks" : 1,
        "plot.marker"        : "o",
        "plot.markersize"    : 3.0,
        "legend.fontsize"    : 9.0,
    }
    fig = gp1.gplot1D( config=config )
    fig.add__plot( xAxis=df1d["zAxis"], yAxis=df1d["Bx"], label="Bx" )
    fig.add__plot( xAxis=df1d["zAxis"], yAxis=df1d["By"], label="By" )
    fig.add__plot( xAxis=df1d["zAxis"], yAxis=df1d["Bz"], label="Bz" )
    fig.set__axis()
    fig.set__legend()
    fig.save__figure()

# ========================================================= #

    colormap__bfield()
```
